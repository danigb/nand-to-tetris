    class SymbolTable {
  constructor () {
    this.table = Object.assign({}, PREDEFINED)
  }
  addEntry (symbol, address) {
    this.table[symbol] = address
    return this
  }
  contains (symbol) {
    return this.getAddress(symbol) !== undefined
  }
  getAddress (symbol) {
    return this.table[symbol]
  }
}

var PREDEFINED = {
  'SP': 0,
  'LCL': 1,
  'ARG': 2,
